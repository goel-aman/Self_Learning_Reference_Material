#include<iostream>
#include<bits/stdc++.h>
using namespace std;

// Min Coins Bottom Up Approach
int minCoinsBU(int N,int coins[],int t){
	int dp[100] = {0};
	
	// Iterate over all states 1 ... N
	for(int n=1;n<=N;n++){
		// Init the current ans as int_max
		dp[n] = INT_MAX;
		for(int i=0;i<t;i++){
			if(n-coins[i] >=0){
				int subprob = dp[n-coins[i]];
				dp[n] = min(dp[n],subprob+1);	
			}
		}
	}
	return dp[N];
}

int main(){
	int N = 15;
	int coins[] = {1,7,10};
	int dp[100] = {0};
	int t = sizeof(coins)/sizeof(int);
	cout<<minCoinsBU(N,coins,t);
	
	return 0;
}

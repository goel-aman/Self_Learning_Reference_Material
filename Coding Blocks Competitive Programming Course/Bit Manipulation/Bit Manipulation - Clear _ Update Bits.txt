#include<iostream>

bool isOdd(int n){
	return (n&1);
}

int getBit(int n,int i){
	int mask = (1<<i);
	int bit = (n&mask) > 0 ? 1 : 0;
	return bit;
}

int setBit(int n,int i){
	int mask = (1<<i);
	int ans = n | mask;
	return ans;
}


void clearBit(int &n,int i){ // setting it to zero
	int mask = ~(1 << i);
	n  = n & mask;
	
}

void updateBit(int &n,int i,int v){
	int mask = ~(1<<i);
	int cleared_n = n&mask;
	n = cleared_n | (v<<i);
}

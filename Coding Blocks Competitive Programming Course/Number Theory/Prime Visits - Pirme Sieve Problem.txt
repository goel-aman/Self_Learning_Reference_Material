#include<bits/stdc++.h>
#define l long long int
using namespace std;

void prime_sieve(l* p){
	/// all the odd position as one
	for(l i=3;i<1000005;i+=2){
		p[i] = 1;
	}
	
	for(l i=3;i<1000005;i+=2){
		if(p[i] == 1){
			for(l j=i*i;j<1000005;j = j+i){
				p[j] = 0;
			}
		}
	}
	// special case 
	p[0] = 0;
	p[1] = 0;
	p[2] = 1;
}


int main(){
	l q;
	cin>>q;
	l * p = new l[1000005];
	for(l i=0;i<1000005;i++){
		p[i] = 0;
	}
	prime_sieve(p);
	
	l* csum = new l[1000005];
	for(l i=0;i<1000000;i++){
		csum[i] = 0;
	}
	for(l i=1;i<1000000;i++){
		csum[i] = csum[i-1] + p[i];
	}
	while(q--){
		l a,b;
		cin>>a>>b;
		cout<<csum[b] - csum[a-1]<<endl;
	}
	return 0;
}

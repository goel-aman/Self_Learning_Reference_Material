#include<iostream>
#include<bits/stdc++.h>
#define l long long int
using namespace std;

l gcd(l n1,l n2){
	if(n1 == 0 ){
		return n2;
	}
	if(n2 == 0){
		return n1;
	}
	
	return gcd(n2,n1%n2);
}

// there is a relation between gcd and lcm of a number gcd*lcm = a*b

int main(){
	
	l n1,n2;
	cin>>n1>>n2;
	
	cout<<gcd(n1,n2)<<endl;
	
	return 0;
}

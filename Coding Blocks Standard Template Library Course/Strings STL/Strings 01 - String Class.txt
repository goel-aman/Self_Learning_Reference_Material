#include<iostream>
#include<string>
using namespace std;


int main(){
	// 
	string s0;
	
	string s1("Hello");
	
	string s2 = "hello world";
	
	string s3(s2);
	
	string s4 = s3;
	
	char a[] = {'a','b','c','\0'};
	
	string s5(a);
	
	cout<<s0<<endl;
	cout<<s1<<endl;
	cout<<s2<<endl;
	cout<<s3<<endl;
	cout<<s4<<endl;
	cout<<s5<<endl;
	if(s0.empty()){
		cout<<"s0 is an empty string"<<endl;
	}
	
	s0.append("I love c++");
	cout<<s0<<endl;
	
	s0 += " and python";
	cout<<s0<<endl;
	
	// Remove / Erase
	cout<<s0.length()<<endl;
	s0.clear();
	cout<<s0.length()<<endl;
	
	// compare the string
	
	s0 = "Apple";
	s1 = "Mango";
	
	cout<<s1.compare(s0)<<endl; // Returns an integer == 0 equal or > <
	
	// Overloaded Operator in c++ > < for string 
	
	if(s1 > s0){
		cout<<"mango is lexographically greater than apple"<<endl;
	}
	
	cout<<s1[0]<<endl;
	
	//find substring
	
	string s = "i want to have apple juice";
	int idx = s.find("apple");
	cout<<idx<<endl;
	
	// Remove a word from the string
	cout<<s<<endl;
	string word = "apple";
	int len = word.length();
	s.erase(idx,len+1);
	cout<<s<<endl;
	
	// iterate over all the character in the string
	for(int i=0;i<s1.length();i++){
		cout<<s[i]<<":";
	}
	cout<<endl;
	for(auto it=s.begin();it != s.end();it++){
		cout<<*it<<",";
	}
	
	// for each loop
	
	for(auto c : s1){
		cout<<c<<".";
	}
	
	for(string::iterator itx=s.begin();itx<s.end();itx++){
		cout<<*itx<<"'";
	}
	return 0;
}

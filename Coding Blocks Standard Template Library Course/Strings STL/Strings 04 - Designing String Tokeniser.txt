#include<iostream>
#include<string>
#include<cstring>

using namespace std;

char* mystrtok(char* str,char delimeter){
	 // string nad single char act as delimeter
	 
	 static char *input = NULL;
	 
	 if(str != NULL){
	 	// we are making the first call
	 	input = str;
	 }
	 
	 if(input == NULL){
	 	return NULL;
	 }
	 
	 // start extracting tokens and store them in a different array
	 char * output = new char[strlen(input) + 1];
	 int i=0;
	 for(;i<input[i] != '\0';i++){
	 	if(input[i] != delimeter){
	 		output[i] = input[i];
		}else{
			output[i] = '\0';
			input = input + i + 1;
			return output;
		}
	 }
	 
	 output[i] = '\0';
	 input = NULL;
	 return output;
}


int main(){
	
	char s[100] = "Today, is a rainy, day";
	
	char* ptr = mystrtok(s,' ');
	cout<<ptr<<endl;
	while(ptr!= NULL){
		ptr= mystrtok(NULL,' ');
		cout<<ptr<<endl;
	}
	return 0;
}

#include<iostream>
#include<bits/stdc++.h>
#include<algorithm>
using namespace std;

int main(){
	
	int arr[] = {1,10,11,9,100};
	int n = sizeof(arr)/sizeof(int);
	// Search --> find
	
	int key = 110;
	auto it = find(arr,arr+n,key);
	cout<<"the address of the key is : "<<it<<endl;
	cout<<"the index of the key is : "<<it - arr<<endl;
	cout<<"the value of the it is "<<(*it)<<endl;
	
	
	// if the value is not present then you will get index as n
	if(it-arr == n){
		cout<<"key is not present"<<endl;
	}
	
	return 0;
}
